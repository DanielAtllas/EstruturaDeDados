		//Insere Final
		int insereFinal(Elem **lista, int x){
			if(vazia(*lista)){
				insereInicio(lista,x);

			}else{
				Elem *no;
				no = (Elem *)malloc(sizeof(Elem));
				no->valor = x;
				Elem *aux;
				aux = *lista;
				while(aux->prox != NULL){
					aux = aux->prox;
				}
				no->prox = NULL;
				aux->prox = no;
			}
			
			return 0;
		}
